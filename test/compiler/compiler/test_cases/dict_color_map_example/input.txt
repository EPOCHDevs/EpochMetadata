# Test dictionary literal support in CardSchemaFilter color_map
src = market_data_source()
signal = src.c > number(value=100).result
status = "ACTIVE" if signal else "INACTIVE"

card_selector_filter(
    card_schema=CardSchemaFilter(
        title="Status Cards",
        icon=Signal,
        select_key="signal#result",
        schemas=[
            CardColumnSchema(
                column_id="status#result",
                slot=Hero,
                render_type=Text,
                color_map={Success: ["ACTIVE"], Error: ["INACTIVE"]}
            )
        ]
    )
)(signal, status)
