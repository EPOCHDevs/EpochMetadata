# Real-world strategy: Track insider purchases as bullish signals
# Insiders (officers, directors, 10%+ owners) buying stock with their own money
# is often a strong signal they believe the stock is undervalued

# Get insider trading data from Form 4 filings
# Filter for open market purchases (P) only - these are most bullish
# Only track significant purchases ($100K+) to filter noise
insider_purchases = insider_trading(
    filing_type="4",
    transaction_code="P",
    min_value=100000,
    timeframe="1D"
)()

# Combine with price action to validate signals
# Get market data to compare insider purchase price vs current price
src = market_data_source()
current_price = src.c

# Calculate if stock is still near insider purchase price
# If insiders bought recently at current levels, stronger signal
price_vs_insider = current_price / insider_purchases.price

# All insider transaction details available as outputs:
# insider_purchases.transaction_date - When trade occurred
# insider_purchases.owner_name - Insider name (CEO, CFO, Director)
# insider_purchases.transaction_code - Transaction type (P=Purchase)
# insider_purchases.shares - Number of shares purchased
# insider_purchases.price - Price per share paid by insider
# insider_purchases.ownership_after - Total shares owned after transaction
