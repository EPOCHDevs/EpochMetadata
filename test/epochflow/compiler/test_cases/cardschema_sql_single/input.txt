# Test card_selector_sql with card_schema (CardSchemaSQL)
src = market_data_source()
short_ma = ema(period=15)(src.c)
long_ma = ema(period=100)(src.c)
bullish = short_ma.result > long_ma.result

card_selector_sql(
    card_schema=CardSchemaSQL(
        title="Bullish Signals",
        icon=Signal,
        sql="SELECT SLOT0 as RESULT0 FROM self WHERE SLOT0 = 1",
        schemas=[]
    )
)(bullish)
