src = market_data_source()

# Quality metrics
ratios = financial_ratios()
low_debt = ratios.debt_equity < 0.7
liquid = ratios.current_ratio > 1.3
high_roe = ratios.roe > 0.15

# Value signal (quality fundamentals)
high_quality = low_debt and liquid and high_roe

# Momentum signal
momentum = roc(period=60)(src.c)
positive_momentum = momentum > 0

# Combined signal (quality + momentum)
entry = high_quality and positive_momentum

trade_signal_executor()(enter_long=entry)
