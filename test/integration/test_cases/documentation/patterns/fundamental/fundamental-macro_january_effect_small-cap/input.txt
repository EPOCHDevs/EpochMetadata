# Section: Fundamental & Macro > January Effect (Small-Cap)
# Type: complete_strategy

src = market_data_source()

# January detection
january = month_of_year(month="January")()
is_january = january.result

# Small-cap momentum (cross-sectional)
returns = roc(period=20)(src.c)
momentum = cs_momentum()(returns)

# Stronger filter in January (top 20 vs top 10)
# More aggressive in January, normal rest of year
entry = conditional_select(
    is_january, top_k(k=20)(momentum),
    top_k(k=10)(momentum)
)

trade_signal_executor()(enter_long=entry)
