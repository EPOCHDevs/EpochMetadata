# Section: Advanced Momentum Indicators > Fisher Transform
# Type: complete_strategy

src = market_data_source()

fisher_result = fisher(period=10)(src.h, src.l)
fisher_val = fisher_result.fisher
fisher_signal = ema(period=3)(fisher_val)

# Extreme reversal signals
fisher_oversold = fisher_val < -2.0
fisher_overbought = fisher_val > 2.0

# Crossover
bullish_cross = crossover(fisher_val, fisher_signal)
bearish_cross = crossunder(fisher_val, fisher_signal)

entry = fisher_oversold and bullish_cross
exit = fisher_overbought or bearish_cross

trade_signal_executor()(enter_long=entry, exit_long=exit)
