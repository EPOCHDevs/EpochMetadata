# Source: strategies/advanced-momentum.md:121
# Section: Advanced Momentum Indicators > Multi-Oscillator Confirmation
# Type: complete_strategy

src = market_data_source()

fisher = fisher(period=10)(src.h, src.l)
uo = ultosc(short_period=7, medium_period=14, long_period=28)(src.h, src.l, src.c)
trix_val = trix(period=14)(src.c)
qstick_val = qstick(period=14)(src.o, src.c)

# Reversal confluence
fisher_os = fisher.fisher < -1.5
fisher_cross = crossover(fisher.fisher, 0)

uo_div = (src.l < min(period=20)(src.l[1])) and (uo.result > uo.result[20])

trix_bullish = trix_val > 0

qstick_acc = qstick_val.result > 0.2

# High-probability setup
reversal = fisher_os and fisher_cross and uo_div
trend_ok = trix_bullish
momentum_ok = qstick_acc

entry = reversal and trend_ok and momentum_ok

# Exit
fisher_ob = fisher.fisher > 2.0
trix_bearish = crossunder(trix_val, 0)

exit = fisher_ob or trix_bearish

trade_signal_executor()(enter_long=entry, exit_long=exit)
