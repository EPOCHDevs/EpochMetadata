# Pattern 3: Comparing against a benchmark index (e.g., SPX)
# Each asset's data automatically provided by market_data_source
src = market_data_source(timeframe="1D")

# Fetch benchmark separately (not in universe, just for comparison)
spx = common_indices(index="SPX", timeframe="1D")()
# Or: spx = indices(ticker="SPX", timeframe="1D")()

# Compare asset performance vs SPX
asset_return = roc(period=20)(src.c)
spx_return = roc(period=20)(spx.c)

# Relative strength: asset outperforming SPX
outperforming = asset_return > spx_return

trade_signal_executor()(enter_long=outperforming)
