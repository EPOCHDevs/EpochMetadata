src = market_data_source(timeframe="1D")
spx = common_indices(index="SPX")

# Calculate factors
momentum = cs_zscore()(roc(period=60)(src.c))
correlation = cs_rolling_corr(window=60)(roc(period=1)(src.c), roc(period=1)(spx.c))

# Compare factors across assets
cs_table_report(agg="last", title="Factor Scores", category="Cross-Sectional")(momentum, correlation)

# Visualize distributions
histogram_chart_report(bins=20, title="Momentum Distribution")(momentum)
