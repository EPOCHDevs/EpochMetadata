src = market_data_source()

# These automatically plot on chart
ema_20 = ema(period=20)(src.c)        # → Line overlay
rsi_val = rsi(period=14)(src.c)       # → Panel below chart

# Mark RSI crossover events with colored badges
cross_above_30 = crossover(rsi_val, 30)
cross_below_70 = crossunder(rsi_val, 70)
flag = cross_above_30 or cross_below_70
label = conditional_select(
    cross_above_30, "RSI Cross Above 30 (Bullish)",
    cross_below_70, "RSI Cross Below 70 (Bearish)")

event_marker(
    event_marker_schema=EventMarkerSchema(
        title="RSI Crossovers",
        icon=TrendUp,
        select_key=SLOT0,
        schemas=[
            CardColumnSchema(
                column_id=SLOT1,
                slot=Hero,
                render_type=Badge,
                color_map={Success: ["RSI Cross Above 30 (Bullish)"], Error: ["RSI Cross Below 70 (Bearish)"]}
            )
        ]
    )
)(flag, label)
