# Market OHLC bars

# Fundamental screening
ratios = financial_ratios(timeframe="1Q")
income = income_statement(timeframe="1Q")
quality = (ratios.roe > 0.15) & (income.net_income > 0)

# Economic regime detection
fed_funds = economic_indicator(category="FedFunds", timeframe="1D")
risk_on = fed_funds.value < 2.0

# Insider sentiment
insider = insider_trading(
    transaction_code="P",
    min_value=100000,
    timeframe="1D"
)()

# Combined signal
signal = quality & risk_on & insider

trade_signal_executor()(enter_long=signal)
